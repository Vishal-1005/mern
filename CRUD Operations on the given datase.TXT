Server.js
const express = require("express");
const mongoose = require("mongoose");
const bodyParser = require("body-parser");
const cors = require("cors");

const app = express();
app.use(cors());
app.use(bodyParser.json());

// MongoDB connection string
mongoose.connect("mongodb+srv://mownika:admin@cluster0.iwolo6u.mongodb.net/testdb?retryWrites=true&w=majority&appName=Cluster0")
  .then(() => console.log(" MongoDB Connected"))
  .catch((err) => console.log(" MongoDB Connection Error:", err));

// Schema
const userSchema = new mongoose.Schema({
  name: String,
  email: String,
  age: Number
});

// Model
const User = mongoose.model("User", userSchema);

// --- CRUD Operations ---

// CREATE
app.post("/users", async (req, res) => {
  const user = new User(req.body);
  await user.save();
  res.send("User Created!");
});

// READ
app.get("/users", async (req, res) => {
  const users = await User.find();
  res.json(users);
});

// UPDATE
app.put("/users/:id", async (req, res) => {
  await User.findByIdAndUpdate(req.params.id, req.body);
  res.send("User Updated!");
});

// DELETE
app.delete("/users/:id", async (req, res) => {
  await User.findByIdAndDelete(req.params.id);
  res.send("User Deleted!");
});

// Start Server
app.listen(3000, () => console.log(" Server running on http://localhost:3000"));

VIVA QUESTIONS : 

Here are **short and clear answers** for each question:

---

### **1. What is CRUD?**

CRUD stands for **Create, Read, Update, Delete**—the four basic operations used in databases and APIs.

---

### **2. What is MongoDB?**

MongoDB is a **NoSQL database** that stores data in flexible, JSON-like documents.

---

### **3. What is Mongoose?**

Mongoose is an **ODM (Object Data Modeling) library** for MongoDB that helps structure and validate data in Node.js.

---

### **4. What is Express.js?**

Express.js is a **Node.js web framework** used to build APIs and server-side applications easily.

---

### **5. What is Postman?**

Postman is a **tool used to test APIs** by sending requests (GET, POST, PUT, DELETE) and viewing responses.

---

### **6. What is a Collection and a Document in MongoDB?**

* **Collection:** A group of related documents (similar to a table).
* **Document:** A single record stored in JSON/BSON format.

---

### **7. How do you connect Node.js with MongoDB?**

You connect using **Mongoose** or the **MongoDB Node.js driver**, usually with a connection string like:
`mongoose.connect("mongodb://localhost:27017/dbname")`.

---

### **8. What is ObjectId?**

ObjectId is a **unique 12-byte identifier** automatically generated by MongoDB for each document.
